#!/bin/bash

MAX=0
MIN=0

for i in {1..10}; do
a=$RANDOM

if [ $i -eq 1 ]; then
MIN=$a
fi

if [ $i -le 9 ]; then
echo -n “$a,”
else
echo $a
fi

if [ $a -lt $MIN ]; then
MIN=$a
fi

if [ $a -gt $MAX ]; then
MAX=$a
fi

done

echo “max number is $MAX”
echo “min number is $MIN”

#!/bin/bash

case $1 in
[0-9])
echo “$1 is a digit.”
;;
[[:lower:]])
echo “$1 is a lower.”
;;
[[:upper:]])
echo “$1 is an upper.”
;;
*)
echo “special character.”
;;
esac
